### モジュール

```
module Math
  def add(x,y)
    x + y
  end
  def average(x,y)
    (x + y) / 2
  end
  def debug()
    puts "math debug"
  end
  
  module_function :add,:average
end

module SumModule
  def debug()
    puts "sum-module debug"
  end
end

class TestModule
  include Math # Mix-in:クラスからモジュールを呼ぶ
  include SumModule
  
  def showAdd(x,y)
    result = add(x,y)
    print "add:",x,"and",y,"is",result,"\n"
  end
  def showAverage(x,y)
    result = average(x,y)
    print "average:",x,"and",y,"is",result,"\n"
  end
  def showDebug()
     debug
  end
end

test = TestModule.new
test.showAdd(10,8)
test.showAverage(100,50)

p TestModule.ancestors
test.showDebug

puts Math.average(70,40)
```
